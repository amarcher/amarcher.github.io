Loop # of Times

loop

n = 0
loop {
  puts 'hi'
  n += 1
  break if n == 5
}

times

(1..10).times { puts 'hi'}

loop

n = 0
for(;;) {
  console.log('hi');
  n++;
  if (n == 5) {break;}
}

Indexed For Loop

manual indexing for loop

arr = [1,2,3,4,5]
for i in 0...arr.length {
  puts arr[i]
}

manual indexing for loop

var arr = [1,2,3,4,5]
for (var i=0; i < arr.length; i++) {
  console.log(arr[i]);
}

Integrated For Loop

hash

hash = {1: "hello"}
hash.each { |elem|
  puts elem
}
# => [1,"hello"]

array

arr = [1,2,3,4,5]
arr.each { |val|
  puts val
}

object

var obj = {1: "hello"}
for (var param in obj) {
  console.log(param);
}
// => 1 (not "1: hello"!)

array

var arr = [1,2,3,4,5]
for (var num in arr) {
  console.log(num);
}

Pre-Test Loops

while

n = 0
while(n < 6) {
  puts n
  n += 1
}

until

n = 0
until(n > 5) {
  puts n
  n += 1
}

while

var n = 0
while (n < 6) {
  console.log(n);
  n++;
}

Post-Test Loops

begin ... end while

n = 0
begin
  puts n
  n += 1
end while n < 6

do ... while

var n = 0
do {
  console.log(n);
  n++;
}
while (n < 6);
